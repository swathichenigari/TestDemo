CREATE TABLE `student` (
  `ID` int NOT NULL AUTO_INCREMENT,  
  `age` int NOT NULL,
  `caste` varchar(255) DEFAULT NULL,
  `dateofBirth` datetime(6) DEFAULT NULL,
  `gender` varchar(255) DEFAULT NULL,
  `mobileNo` bigint NOT NULL,
  `name` varchar(255) NOT NULL,
  `relegion` varchar(255) DEFAULT NULL,
  `address_aid` int NOT NULL,
  PRIMARY KEY (`ID`),
  CONSTRAINT `FKdwi17jualiaokc9cti5ilffha` FOREIGN KEY (`address_aid`) REFERENCES `address` (`aid`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;


CREATE TABLE `marks` (
  `ID` int NOT NULL AUTO_INCREMENT,
  `marksObtained` int NOT NULL,
  `semester` int NOT NULL,
  `studentId` int NOT NULL,
  `subjectCode` varchar(255) DEFAULT NULL,
  `totalMarks` int NOT NULL,
  PRIMARY KEY (`ID`)
) ENGINE=InnoDB AUTO_INCREMENT=3 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `address` (
  `aid` int NOT NULL AUTO_INCREMENT,
  `city` varchar(255) DEFAULT NULL,
  `district` varchar(255) DEFAULT NULL,
  `houseNo` varchar(255) DEFAULT NULL,
  `pincode` int NOT NULL,
  `state` varchar(255) DEFAULT NULL,
  PRIMARY KEY (`aid`)
) ENGINE=InnoDB AUTO_INCREMENT=3 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;
